function l { Get-ChildItem -Force $args }

# Fuzzy history search with fzf
function Invoke-FzfHistory {
    # Get PowerShell history
    $hist = Get-History | Select-Object -ExpandProperty CommandLine

    # Call fzf
    $selection = $hist | fzf --prompt "History> " --height 40%

    if ($selection) {
        # Insert selected command into shell
        [System.Console]::WriteLine($selection)
        Invoke-Expression $selection
    }
}

# Bind Ctrl-R to fzf history using a scriptblock
Set-PSReadLineKeyHandler -Chord Ctrl+r -ScriptBlock {
    Invoke-FzfHistory
}

Remove-PSReadLineKeyHandler -Chord 'Ctrl+Spacebar'
